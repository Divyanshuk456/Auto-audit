<?php

function scanPrinter($conn,$line,$board,$folder){

    if(!is_dir($folder)){
        markOffline($conn,$line,$board);
        return;
    }

    $latest=getLatestLog($folder);

    if(!$latest){
        markOffline($conn,$line,$board);
        return;
    }

    $modified=date(
        "Y-m-d H:i:s",
        filemtime($latest)
    );

    $stmt=$conn->prepare(
    "SELECT last_file
     FROM sync_status
     WHERE line_name=? AND board=?");

    $stmt->bind_param("ss",$line,$board);
    $stmt->execute();

    $res=$stmt->get_result();

    $last="";

    if($r=$res->fetch_assoc()){
        $last=$r["last_file"];
    }

    if($last==$latest){
        return;
    }

    $log=parseLog($latest);

    updateCache(
        $conn,
        $line,
        $board,
        $log,
        $latest,
        $modified
    );

    $stmt=$conn->prepare(
    "REPLACE INTO sync_status
    (line_name,board,last_file,last_modified)
    VALUES(?,?,?,?)");

    $stmt->bind_param(
        "ssss",
        $line,
        $board,
        $latest,
        $modified
    );

    $stmt->execute();

}

?>
