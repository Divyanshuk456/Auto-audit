<script>
document.getElementById('line').addEventListener('change', function(){

    let line=this.value;
    document.getElementById('saveLine').value=line;

    if(line==="") return;

    fetch("autofill.php?line="+encodeURIComponent(line))
    .then(r=>r.json())
    .then(data=>{

        ["TOP","BOTTOM"].forEach(function(board){

            if(!data.printers[board]) return;

            let p=data.printers[board].data;

            document.getElementById(board+"_Thickness").innerText=p.Thickness ?? "-";
            document.getElementById(board+"_Temperature").innerText=p.Temperature ?? "-";
            document.getElementById(board+"_Pressure").innerText=p.Pressure ?? "-";

        });

    });

});
</script>
