<select id="line">
<option value="">Select Line</option>
<option value="SMD-01">SMD-01</option>

<option value="SMD-03">SMD-03</option>
</select>

PCB Thickness
<input type="text" id="pcbThickness">

Pressure
<input type="text" id="pressure">

Temperature
<input type="text" id="temperature">

<script>
$("#line").change(function(){

var selectedLine=$(this).val();

if(selectedLine==""){ return; }

$.ajax({

url:"autofill.php",
type:"POST",
dataType:"json",
data:{line:selectedLine},

success:function(response){

$("#pcbThickness").val(response.pcbThickness);
$("#pressure").val(response.pressure);
$("#temperature").val(response.temperature);

},

error:function(xhr,status,error){

console.log(xhr.responseText);
alert("Unable to fetch printer log.");

}

});

});
</script>

</body>
</html>

==========================
File: autofill.php
==========================

<?php

header("Content-Type: application/json");

$line=$_POST['line'] ?? '';

switch($line){

case "SMD-01":
echo json_encode([
"pcbThickness"=>"1.60",
"pressure"=>"40",
"temperature"=>"27"
]);
break;

case "SMD-02":
echo json_encode([
"pcbThickness"=>"1.55",
"pressure"=>"39",
"temperature"=>"28"
]);
break;

default:
echo json_encode([
"pcbThickness"=>"",
"pressure"=>"",
"temperature"=>""
]);

}

?>
